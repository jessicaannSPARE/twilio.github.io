def authorization_documents(sid=:unset)
  if sid.nil?
    raise ArgumentError, 'sid cannot be nil'
  elsif sid == :unset
    @authorization_documents ||= AuthorizationDocumentList.new self
  else
    AuthorizationDocumentContext.new(self, sid)
  end
end
