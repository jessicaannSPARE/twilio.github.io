def initialize(version, sid)
  super(version)

  # Path Solution
  @solution = {
      sid: sid,
  }
  @uri = "/Rooms/#{@solution[:sid]}"

  # Dependents
  @recordings = nil
end

def fetch

      'GET',
      @uri,
      params,
  )

  RoomInstance.new(
      @version,
      payload,
      sid: @solution[:sid],
  )
end

def recordings(sid=:unset)
  raise ArgumentError, 'sid cannot be nil' if sid.nil?

  if sid != :unset
    return RoomRecordingContext.new(
        @version,
        @solution[:sid],
        sid,
    )
  end

  unless @recordings
    @recordings = RoomRecordingList.new(
        @version,
        room_sid: @solution[:sid],
    )
  end

  @recordings
end

def update(status: nil)
  data = Twilio::Values.of({
      'Status' => status,
  })

  payload = @version.update(
      'POST',
      @uri,
      data: data,
  )

  RoomInstance.new(
      @version,
      payload,
      sid: @solution[:sid],
  )
end
