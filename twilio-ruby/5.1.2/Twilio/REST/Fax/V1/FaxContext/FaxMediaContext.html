def initialize(version, fax_sid, sid)
  super(version)

  # Path Solution
  @solution = {
      fax_sid: fax_sid,
      sid: sid,
  }
  @uri = "/Faxes/#{@solution[:fax_sid]}/Media/#{@solution[:sid]}"
end

def fetch
  params = Twilio::Values.of({})

  payload = @version.fetch(
      'GET',
      @uri,
      params,
  )

  FaxMediaInstance.new(
      @version,
      payload,
      fax_sid: @solution[:fax_sid],
      sid: @solution[:sid],
  )
end
