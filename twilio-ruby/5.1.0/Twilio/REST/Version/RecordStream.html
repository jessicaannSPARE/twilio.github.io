def each
  current_record = 0
  current_page = 1

  while @page
    @page.each do |record|
      yield record
      current_record += 1
      return nil if @limit && @limit <= current_record
    end

    return nil if @page_limit && @page_limit <= current_page

    @page = @page.next_page
    current_page += 1
  end
end
