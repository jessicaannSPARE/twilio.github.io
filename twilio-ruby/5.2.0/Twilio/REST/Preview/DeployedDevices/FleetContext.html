def initialize(version, sid)
  super(version)

  # Path Solution
  @solution = {
      sid: sid,
  }
  @uri = "/Fleets/#{@solution[:sid]}"

  # Dependents
  @devices = nil
  @deployments = nil
  @certificates = nil
  @keys = nil
end

def certificates(sid=:unset)
  raise ArgumentError, 'sid cannot be nil' if sid.nil?

  if sid != :unset
    return CertificateContext.new(
        @version,
        @solution[:sid],
        sid,
    )
  end

  unless @certificates
    @certificates = CertificateList.new(
        @version,
        fleet_sid: @solution[:sid],
    )
  end

  @certificates
end

def update(friendly_name: :unset, default_deployment_sid: :unset)
  data = Twilio::Values.of({
      'FriendlyName' => friendly_name,
      'DefaultDeploymentSid' => default_deployment_sid,
  })

  payload = @version.update(
      'POST',
      @uri,
      data: data,
  )

  FleetInstance.new(
      @version,
      payload,
      sid: @solution[:sid],
  )
end
